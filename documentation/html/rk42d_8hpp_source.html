73  /*
74  Initialize stepper. Stepper will use gpu with id deviceId.
75  Stepper will work with nSystems system simultaneously.
76 
77  One system has dimx * dimy elements and occupies pitchx * dimy bytes. the memory of a system can be padded such that dimx < pitchx / sizeof(double).
78  Thus, including the padding a system has systemsize elements.
79 
80  ode_system_size is the combined total size of all systems, which is systemsize * nSystems
81  */
82  void init(int deviceId, size_t pitchx, size_t dimx, size_t dimy, size_t nSystems, size_t systemsize, size_t ode_system_size, void* userdata,
83  void (*stepfunc)(const size_t* const activeIndices, size_t nIndices,
84  const double* const t, double* const y, double* const y_derived, void* userdata));
85 
86  /*
87  activeIndicesH and activeIndicesD store the indices of paths which participate in the current step.
88  For path activeIndicesD[i], perform Runge-Kutta step for time t[activeIndicesD[i]] with step size h[activeIndicesD[i]]
89 
90  All pointers but activeIndicesH point to GPU memory.
91 
92  length(activeIndicesH) = length(activeIndicesD) = nIndices
93  length(t) = length(h) = nSystems
94  length(y) = ode_system_size
95  */
96  void step(const size_t* activeIndicesH, const size_t* activeIndicesD, size_t nIndices, double* y, const double* t, const double* h);
97 
98  /*
99  activeIndicesH and activeIndicesD store the indices of paths which participate in the current step.
100  For path activeIndicesD[i], perform Runge-Kutta step for time t[activeIndicesD[i]] with step size h[activeIndicesD[i]]
101  and performs error estimation
102 
103  All pointers but activeIndicesH point to GPU memory.
104 
105  length(activeIndicesH) = length(activeIndicesD) = nIndices
106  length(t) = length(h) = nSystems
107  length(y) = length(yerr) = length(dydt_in) = length(dydt_out) = ode_system_size
108  */
109  void step_apply(const size_t* activeIndicesH, const size_t* activeIndicesD, size_t nIndices, const double* t, const double* h,
110  double* y, double* yerr, const double* dydt_in, double* dydt_out);
