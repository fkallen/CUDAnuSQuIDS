205  template<class Func>
206  void setInitialFlux(Func fluxGenerator, nusquids::Basis fluxbasis_){
207  initialFlux.resize(getNumPaths() * GetNumRho() * GetNumE() * GetNumNeu());
208 
209  size_t k = 0;
210  for(int index_cosine = 0; index_cosine < getNumPaths(); index_cosine++){
211  for(int index_rho = 0; index_rho < GetNumRho(); index_rho++){
212  for(int index_energy = 0; index_energy < GetNumE(); index_energy++){
213  for(int flv = 0; flv < GetNumNeu(); flv++){
214  initialFlux[k++] = fluxGenerator(index_cosine, index_rho, index_energy, flv);
215  }
216  }
217  }
218  }
219 
220  fluxbasis = fluxbasis_;
221  }
