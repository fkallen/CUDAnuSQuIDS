220  void solveFixed(){
221 
222  size_t nActiveSystemsTimeloop = nSystems;
223 
224  for(size_t i = 0; i < nSystems; i++){
225  tnowH.get()[i] = t_begin[i];
226  hstepH.get()[i] = (t_end[i] - t_begin[i]) / (double)nSteps;
227 
228  int sign = 0;
229 
230  if (hstepH.get()[i] > 0.0){
231  sign = 1;
232  }else{
233  sign = -1;
234  }
235  if (sign * (t_end[i] - t_begin[i]) < 0.0){
236  printf("integration limits and/or step direction not consistent\n");
237  stats[i].status = Status::failure;
238  }
239  if (sign * (t_end[i] - t_begin[i]) > 0.0){
240  activeSystemsTimeloopH.get()[i] = i;
241  }
242  }
243  cudaMemcpyAsync(activeSystemsTimeloopD.get(), activeSystemsTimeloopH.get(), sizeof(size_t) * nActiveSystemsTimeloop, H2D, copyStreams[0]); CUERR;
244  cudaMemcpyAsync(tnowD.get(), tnowH.get(), sizeof(double) * nActiveSystemsTimeloop, H2D, copyStreams[1]); CUERR;
245  cudaMemcpyAsync(hstepD.get(), hstepH.get(), sizeof(double) * nActiveSystemsTimeloop, H2D, copyStreams[2]); CUERR;
246 
247  if(showProgress)
248  updateProgress(0.0, 1.0);
249 
250  cudaStreamSynchronize(copyStreams[0]); CUERR;
251  cudaStreamSynchronize(copyStreams[1]); CUERR;
252  cudaStreamSynchronize(copyStreams[2]); CUERR;
253 
254  std::vector<size_t> failedSystems;
255  std::vector<size_t> newActiveSystems(nSystems);
256 
257  bool activeSystemsListNeedsUpdate = false;
258 
259  for(unsigned int step = 0; step < nSteps && nActiveSystemsTimeloop > 0 && step < nSteps; step++){
260 
261  cudaMemcpyAsync(y0evolve.get(), ode_system, sizeof(double) * ode_system_size, D2D, copyStreams[0]); CUERR;
262  cudaStreamSynchronize(copyStreams[0]); CUERR;
263 
264  stepfunc(activeSystemsTimeloopD.get(), nActiveSystemsTimeloop, tnowD.get(), ode_system, dydt_in.get(), userdata);
265 
266  stepper.step_apply(activeSystemsTimeloopH.get(),activeSystemsTimeloopD.get(), nActiveSystemsTimeloop, tnowD.get(), hstepD.get(),
267  ode_system, error.get(), dydt_in.get(), dydt_out.get());
268 
269  cudaMemcpyAsync(holdD.get(), hstepD.get(), sizeof(double) * nActiveSystemsTimeloop, D2D, updateStream1); CUERR;
270 
271  callSolver2dHadjustKernelAsync(
272  hadjustRetvalsD.get(), Stepper::getOrder(),
273  activeSystemsTimeloopD.get(), nActiveSystemsTimeloop,
274  ode_system, error.get(), dydt_out.get(),
275  holdD.get(), systemsize, eps_abs, eps_rel, dimx, dimy, pitchx,
276  dim3(1, nActiveSystemsTimeloop, 1), dim3(256, 1, 1), updateStream1);
277 
278  cudaMemcpyAsync(hadjustRetvalsH.get(), hadjustRetvalsD.get(), sizeof(int) * nActiveSystemsTimeloop, D2H, updateStream1); CUERR;
279  cudaStreamSynchronize(updateStream1); CUERR;
280 
281  for(size_t i = 0; i < nActiveSystemsTimeloop; i++){
282  const size_t si = activeSystemsTimeloopH.get()[i];
283  if(hadjustRetvalsH.get()[si] == -1){
284  /* system i failed */
285  cudaMemcpyAsync(ode_system + si * systemsize,
286  y0evolve.get() + si * systemsize,
287  sizeof(double) * systemsize,
288  D2D,
289  copyStreams[i % nCopyStreams]); CUERR;
290 
291  stats[si].status = Status::failure;
292 
293  // insert si into sorted failedSystems vector
294  const auto upperbound = std::upper_bound(failedSystems.begin(), failedSystems.end(), si);
295  if(upperbound == failedSystems.cend())
296  failedSystems.push_back(si);
297  else
298  failedSystems.insert(upperbound, si);
299 
300  activeSystemsListNeedsUpdate = true;
301  }else{
302  /* everything is good */
303  tnowH.get()[si] += hstepH.get()[si];
304 
305  stats[si].steps++;
306  }
307  }
308  cudaMemcpyAsync(tnowD.get(), tnowH.get(), sizeof(double) * nActiveSystemsTimeloop, H2D, copyStreams[0]); CUERR;
309 
310  for(size_t i = 0; i < nCopyStreams; i++)
311  cudaStreamSynchronize(copyStreams[i]); CUERR;
312 
313  if(activeSystemsListNeedsUpdate){
314  // remove every element in failedSystems from activeSystems
315  std::set_difference (activeSystemsTimeloopH.get(),
316  activeSystemsTimeloopH.get() + nActiveSystemsTimeloop,
317  failedSystems.begin(),
318  failedSystems.end(),
319  newActiveSystems.begin());
320 
321  nActiveSystemsTimeloop = nSystems - failedSystems.size();
322 
323  std::copy(newActiveSystems.begin(), newActiveSystems.end(), activeSystemsTimeloopH.get());
324  cudaMemcpyAsync(activeSystemsTimeloopD.get(),
325  activeSystemsTimeloopH.get(),
326  sizeof(size_t) * nActiveSystemsTimeloop,
327  H2D,
328  copyStreams[0]); CUERR;
329  cudaStreamSynchronize(copyStreams[0]); CUERR;
330 
331  activeSystemsListNeedsUpdate = false;
332  }
333 
334  if(showProgress)
335  updateProgress((double)step, nSteps);
336  }
337 
338  if(showProgress)
339  updateProgress(1.0, 1.0);
340  }

342  void solveAdaptive(){
343  size_t nActiveSystemsTimeloop = 0;
344  size_t nActiveSystemsHloop = 0;
345 
346  std::vector<int> signHost(nSystems, 0);
347  std::vector<bool> final_stepHost(nSystems, 0);
348  std::vector<double> dt(nSystems, 0);
349 
350  // this is used to determine current progress
351  double totalTEnd = 0.0;
352 
353  for(size_t i = 0; i < nSystems; i++){
354 
355  hstepH.get()[i] = h;
356  tnowH.get()[i] = t_begin[i];
357 
358  if (hstepH.get()[i] > 0.0){
359  signHost[i] = 1;
360  }else{
361  signHost[i] = -1;
362  }
363  if (signHost[i] * (t_end[i] - t_begin[i]) < 0.0){
364  printf("system %lu, integration limits and/or step direction not consistent\n", i);
365  stats[i].status = Status::failure;
366  }
367  // make a list of systems which need to be calculated, i.e [0, 1, 2, ..., nSystems-1]
368  if (signHost[i] * (t_end[i] - t_begin[i]) > 0.0){
369  activeSystemsTimeloopH.get()[nActiveSystemsTimeloop] = i;
370  nActiveSystemsTimeloop++;
371  }
372 
373  totalTEnd += t_end[i];
374  }
375 
376  cudaMemcpyAsync(tnowD.get(), tnowH.get(), sizeof(double) * nSystems, H2D, copyStreams[1]); CUERR;
377  cudaMemcpyAsync(hstepD.get(), hstepH.get(), sizeof(double) * nSystems, H2D, copyStreams[2]); CUERR;
378 
379  cudaStreamSynchronize(copyStreams[1]); CUERR;
380  cudaStreamSynchronize(copyStreams[2]); CUERR;
381 
382  bool isFirstIter = true;
383 
384  if(showProgress)
385  updateProgress(0.0, 1.0);
386 
387  // loop until every system is calculated
388  while(nActiveSystemsTimeloop > 0){
389 
390  std::copy(tnowH.get(), tnowH.get() + nSystems, t0H.get());
391  cudaMemcpyAsync(t0D.get(), tnowD.get(), sizeof(double) * nSystems, D2D, copyStreams[0]); CUERR;
392 
393  std::copy(hstepH.get(), hstepH.get() + nSystems, h0H.get());
394  cudaMemcpyAsync(h0D.get(), hstepD.get(), sizeof(double) * nSystems, D2D, copyStreams[1]); CUERR;
395 
396  // calculate remaining time for each active system
397  for(size_t i = 0; i < nActiveSystemsTimeloop; i++){
398  const size_t si = activeSystemsTimeloopH.get()[i];
399  dt[si] = t_end[si] - t0H.get()[si];
400  }
401 
402  // copy each active system to y0evolve.
403  for(size_t i = 0; i < nActiveSystemsTimeloop; i++){
404 
405  // group copy operations of successive systems into single copy operations
406  size_t begin = activeSystemsTimeloopH.get()[i];
407  size_t copyBatch = 1;
408  while(i+1 < nActiveSystemsTimeloop && activeSystemsTimeloopH.get()[i+1] == activeSystemsTimeloopH.get()[i] + 1){
409  copyBatch++;
410  i++;
411  }
412 
413  cudaMemcpyAsync(y0evolve.get() + begin * systemsize,
414  ode_system + begin * systemsize,
415  sizeof(double) * systemsize * copyBatch,
416  D2D,
417  copyStreams[i % nCopyStreams]); CUERR;
418  }
419 
420  cudaMemcpyAsync(activeSystemsTimeloopD.get(),
421  activeSystemsTimeloopH.get(),
422  sizeof(size_t) * nActiveSystemsTimeloop,
423  H2D,
424  copyStreams[0]);
425 
426  if(isFirstIter){
427  for(size_t i = 0; i < nCopyStreams; i++)
428  cudaStreamSynchronize(copyStreams[i]); CUERR;
429 
430  stepfunc(activeSystemsTimeloopD.get(), nActiveSystemsTimeloop, t0D.get(), ode_system, dydt_in.get(), userdata);
431 
432  isFirstIter = false;
433  }else{
434  // in other iterations we can reuse the data which was needed by step size control
435 
436  // copy the data of active systems to dydt_in
437  for(size_t i = 0; i < nActiveSystemsTimeloop; i++){
438 
439  size_t begin = activeSystemsTimeloopH.get()[i];
440  size_t copyBatch = 1;
441  while(i+1 < nActiveSystemsTimeloop && activeSystemsTimeloopH.get()[i+1] == activeSystemsTimeloopH.get()[i] + 1){
442  copyBatch++;
443  i++;
444  }
445 
446  cudaMemcpyAsync(dydt_in.get() + begin * systemsize,
447  dydt_out.get() + begin * systemsize,
448  sizeof(double) * systemsize * copyBatch,
449  D2D,
450  copyStreams[i % nCopyStreams]); CUERR;
451  }
452  for(size_t i = 0; i < nCopyStreams; i++)
453  cudaStreamSynchronize(copyStreams[i]); CUERR;
454  }
455 
456  std::copy(activeSystemsTimeloopH.get(), activeSystemsTimeloopH.get() + nSystems, activeSystemsHloopH.get());
457 
458  nActiveSystemsHloop = nActiveSystemsTimeloop;
459 
460  // "h loop". loop until every step was calculated without decreasing the step size h
461  do{
462 
463  /*
464  For each active system, decrease step size if neccessary so we won't get beyond t_end.
465  If this decrease is needed, we know this will be the last step, assuming the step succeeds
466  */
467 
468  for(size_t i = 0; i < nActiveSystemsHloop; i++){
469 
470  const size_t si = activeSystemsHloopH.get()[i];
471 
472  if ((dt[si] >= 0.0 && h0H.get()[si] > dt[si]) || (dt[si] < 0.0 && h0H.get()[si] < dt[si])){
473  h0H.get()[si] = dt[si];
474  final_stepHost[si] = true;
475  }else{
476  final_stepHost[si] = false;
477  }
478  stats[si].steps++;
479  holdH.get()[si] = h0H.get()[si];
480  }
481 
482  cudaMemcpyAsync(h0D.get(),
483  h0H.get(),
484  sizeof(double) * nSystems,
485  H2D,
486  copyStreams[0]); CUERR;
487 
488  cudaMemcpyAsync(activeSystemsHloopD.get(),
489  activeSystemsHloopH.get(),
490  sizeof(size_t) * nActiveSystemsHloop,
491  H2D,
492  copyStreams[1]); CUERR;
493 
494  cudaStreamSynchronize(copyStreams[0]); CUERR;
495  cudaStreamSynchronize(copyStreams[1]); CUERR;
496 
497  stepper.step_apply(activeSystemsHloopH.get(), activeSystemsHloopD.get(),
498  nActiveSystemsHloop, t0D.get(), h0D.get(),
499  ode_system, error.get(), dydt_in.get(), dydt_out.get());
500 
501  for(size_t i = 0; i < nActiveSystemsHloop; i++){
502  const size_t si = activeSystemsHloopH.get()[i];
503 
504  if (final_stepHost[si]){
505  tnowH.get()[si] = t_end[si];
506  }else{
507  tnowH.get()[si] = t0H.get()[si] + h0H.get()[si];
508  }
509  }
510 
511  cudaMemcpyAsync(tnowD.get(), tnowH.get(), sizeof(double) * nSystems, H2D, copyStreams[1]); CUERR;
512  cudaStreamSynchronize(copyStreams[1]); CUERR;
513 
514  callSolver2dHadjustKernelAsync(
515  hadjustRetvalsD.get(), Stepper::getOrder(),
516  activeSystemsHloopD.get(), nActiveSystemsHloop,
517  ode_system, error.get(), dydt_out.get(),
518  h0D.get(), systemsize, eps_abs, eps_rel, dimx, dimy, pitchx,
519  dim3(1, nSystems, 1), dim3(256, 1, 1), updateStream1);
520 
521  cudaMemcpyAsync(hadjustRetvalsH.get(), hadjustRetvalsD.get(), sizeof(int) * nSystems, D2H, updateStream1);
522  cudaMemcpyAsync(h0H.get(), h0D.get(), sizeof(double) * nSystems, D2H, updateStream1);
523 
524  cudaStreamSynchronize(updateStream1); CUERR;
525 
526  int nActiveSystemsHloopNew = 0;
527  std::vector<size_t> activeSystemsHloopHNew(nSystems, 0);
528 
529  /* for each system check if the step size was decreased*/
530  for(size_t i = 0; i < nActiveSystemsHloop; i++){
531 
532  const size_t si = activeSystemsHloopH.get()[i];
533 
534  if (hadjustRetvalsH.get()[i] == -1)
535  {
536  const double t_curr = tnowH.get()[si];
537  const double t_next = tnowH.get()[si] + h0H.get()[si];
538 
539  if (fabs (h0H.get()[si]) < fabs (holdH.get()[si]) && t_next != t_curr)
540  {
541  // reset system
542  cudaMemcpyAsync(ode_system + si * systemsize,
543  y0evolve.get() + si * systemsize,
544  sizeof(double) * systemsize,
545  D2D,
546  copyStreams[i % nCopyStreams]); CUERR;
547 
548  stats[si].repeats++;
549 
550  // add system id to active systems for next try
551  activeSystemsHloopHNew[nActiveSystemsHloopNew] = si;
552  nActiveSystemsHloopNew++;
553  }
554  else
555  {
556  stats[si].status = Status::failure;
557  }
558  }else{
559  /* no change, or increase, everything is good */
560  }
561 
562  }
563 
564  std::copy(activeSystemsHloopHNew.begin(), activeSystemsHloopHNew.end(), activeSystemsHloopH.get());
565  nActiveSystemsHloop = nActiveSystemsHloopNew;
566 
567  for(size_t i = 0; i < nCopyStreams; i++)
568  cudaStreamSynchronize(copyStreams[i]); CUERR;
569
